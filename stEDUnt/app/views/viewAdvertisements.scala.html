@(student: Student, studAdvertisements: List[StudentAdvertisement], tutAdvertisements: List[TutorAdvertisement]) @main("Start") {

All Advertisements

Logged in as @student.name

Student Advertisement

@for(index <- 0 until studAdvertisements.size){

Advertisement @index

Student: @studAdvertisements(index).getStudent().name
Studies: @studAdvertisements(index).getStudies()
Description: @studAdvertisements(index).getDescription() }

Tutor Advertisement

@for(index <- 0 until tutAdvertisements.size){

Advertisement @index

Student: @tutAdvertisements(index).getStudent().name
Studies: @tutAdvertisements(index).getStudies()
Description: @tutAdvertisements(index).getDescription()
Price: @tutAdvertisements(index).getPrice() }
}
